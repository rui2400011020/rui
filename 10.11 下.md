m,n,p,q=map(int,input().split())
mx1=[]
mx2=[]
answer=[]
for _ in range(m):
    mx1.append(list(map(int,input().split())))
for _ in range(p):
    mx2.append(list(map(int,input().split())))
for i in range(m-p+1):
    list=[]
    for j in range(n-q+1):
        sum=0
        for s in range(p):
            for t in range(q):
                sum+=mx1[i+s][j+t]*mx2[s][t]
        list.append(sum)
    answer.append(list)
for x in answer:
    print(*x)
