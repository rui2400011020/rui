```
n=int(input())
x=[]
for _ in range(n):
    a,b=map(int,input().split())
    x.append([a,b])
x.sort()
t=0
for i in range(n-1):
    if x[i][1]<x[i+1][1]:
        t+=1
    else:
        print('Happy Alex')
        break
if t==n-1:
    print('Poor Alex')
```
