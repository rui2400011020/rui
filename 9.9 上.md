a=[]
t=0
for i in range(2,300):
    t=t+1/i
    a.append(t)
b=[]
while True:
    c=float(input())
    if c==0:
        break
    else:
        b.append(c)
for i in b:
    for j in range(0,300):
        if a[j]>=i:
            print(f'{str(j+1)} card(s)')
            break
