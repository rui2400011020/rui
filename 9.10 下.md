n=int(input())
a=[]
while n>1:
    for i in range(2,n+1):
        if n%i==0:
            a.append(i)
            n=n//i
            break
b=set(a)
A=len(a)
B=len(b)
if A>B:
    print(0)
elif A==B:
    if A%2==0:
        print(1)
    else:
        print(-1)
