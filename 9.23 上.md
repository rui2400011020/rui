a,b=map(int,input().split())
c=[]
def task(i):
    a=i//100
    i=i%100
    b=i//10
    c=i%10
    if 100*a+10*b+c==a**3+b**3+c**3:
        return True
    else:
        return False
for i in range(a,b+1):
    if task(i):
        c.append(i)
if c:
    print(*c)
else:
    print('NO')
