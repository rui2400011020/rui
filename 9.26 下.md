t=int(input())
N=[]
for i in range(t):
    N.append(int(input()))
for n in N:
    a=[1]*n
    for i in range(1,n+1):
        for j in range(1,n+1):
            if j%i==0:
                a[j-1]=-a[j-1]
    print(a.count(-1))
