```
n=int(input())
a=list(map(int,input().split()))
b=[]
for i in range(1,n+1):
    b.append(i)
c=[]
for i in a:
    if i<=n:
        b.remove(i)
    else:
        c.append(i)
b.sort()
c.sort()
print(*b)
print(*c)
```
