position=input()
s=input()
t=''
aa="z x c v b n m , . / a s d f g h j k l ; ' q w e r t y u i o p [ ]"
a=list(aa.split())
def R(x):
    for i in range(33):
        if a[i]==x:
            return a[i-1]
def L(x):
    for i in range(33):
        if a[i]==x:
            return a[i+1]
if position=='R':
    for i in range(len(s)):
        t=t+R(s[i])
if position=='L':
    for i in range(len(s)):
        t=t+L(s[i])
print(t)
