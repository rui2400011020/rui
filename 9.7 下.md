n=int(input())
answer=0
for i in range(n):
    s=input()
    t=len(s)
    number=0
    for i in range(t):
        if s[i]=='#':
            number+=1
    number=number//6
    for i in range(t-6):
        if s[i]==s[i+1]==s[i+2]==s[i+4]==s[i+5]==s[i+6]=='#' and s[i+3]==' ':
            number-=1
    answer+=number
print(answer)
