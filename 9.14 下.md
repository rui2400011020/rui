def f(x,y):
    t=[]
    for i in range(1,min(x,y)+1):
        if x%i==y%i==0:
            t.append(i)
    return max(t)
a=[]
b=[]
while True:
    try:
        x,y=map(int,input().split())
        a.append(x)
        b.append(y)
    except EOFError:
        break
for i in range(len(a)):
    print(f(a[i],b[i]))
