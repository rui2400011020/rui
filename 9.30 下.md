num=int(input())
L=[]
N=[]
P=[]
for _ in range(num):
    l,n=map(int,input().split())
    place=list(map(int,input().split()))
    L.append(l)
    N.append(n)
    P.append(place)
for i in range(num):
    l=L[i]
    n=N[i]
    place=P[i]
    short=[]
    long=[]
    for ant in place:
        short.append(min(ant,l-ant))
        long.append(max(ant,l-ant))
    print(max(short),max(long))
