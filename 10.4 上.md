```
n=int(input())
Line=list(map(int,input().split()))
Line.sort()
line=[]
for i in Line:
    line.append([i,1])
s=0
for i in range(n):
    if s>line[i][0]:
        line[i][1]=0
    s+=line[i][0]*line[i][1]
answer=[]
for i in range(n):
    answer.append(line[i][1])
print(answer.count(1))
```
