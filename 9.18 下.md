a,b=map(int,input().split())
A=[]
for i in range(a):
    t=list(map(int,input().split()))
    A.append(t)
c,d=map(int,input().split())
B=[]
for i in range(c):
    t=list(map(int,input().split()))
    B.append(t)
e,f=map(int,input().split())
C=[]
for i in range(e):
    t=list(map(int,input().split()))
    C.append(t)
if b==c and a==e and d==f:
    D=[]
    for i in range(a):
        x=[]
        for j in range(d):
            y=0
            for z in range(b):
                y=y+A[i][z]*B[z][j]
            y=y+C[i][j]
            x.append(y)
        D.append(x)
    for sx in D:
        print(*sx)
else:
    print('Error!')
