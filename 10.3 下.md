n=int(input())
S=[]
aa=[]
bb=[]
AA=[]
BB=[]
for _ in range(n):
    S.append(int(input()))
    aa.append(int(input()))
    AA.append(list(map(int,input().split())))
    bb.append(int(input()))
    BB.append(list(map(int,input().split())))
for i in range(n):
    s=S[i]
    A=AA[i]
    B=BB[i]
    answer=0
    for x in A:
        answer+=B.count(s-x)
    print(answer)
