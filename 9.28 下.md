N=[]
M=[]
while True:
    n,m=map(int,input().split())
    if n==0 and m==0:
        break
    else:
        N.append(n)
        M.append(m)
g=len(N)
for i in range(g):
    n=N[i]
    m=M[i]
    flag=[]
    for i in range(1,n+1):
        flag.append(i)
    for _ in range(n-1):
        x=m%len(flag)
        if x==0:
            x=len(flag)
        flag=flag[x:]+flag[:x-1]
    print(flag[0])
