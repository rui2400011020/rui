def task(n):
    flag=[1]*(n+1)
    prime=[]
    for i in range(2,n+1):
        if flag[i]:
            prime.append(i)
        for j in prime:
            if j*i>n:
                break
            flag[i*j]=0
            if i%j==0:
                break
    return flag
S=int(input())
flag=task(S)
flag[0]=0
flag[1]=0
if S%2==1:
    print(2*(S-2))
else:
    for i in range(S//2,S-2):
        if flag[i] and flag[S-i]:
            print(i*(S-i))
            break
