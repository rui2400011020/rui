import math
n=int(input())
A=[]
B=[]
C=[]
for i in range(n):
    a,b,c=map(float,input().split())
    A.append(a)
    B.append(b)
    C.append(c)
for i in range(n):
    a=A[i]
    b=B[i]
    c=C[i]
    t=b*b-4*a*c
    if t==0:
        x1=(-b)/(2*a)
        x2=(-b)/(2*a)
        if x1==-0:
            x1=0
        if x2==-0:
            x2=0
        print(f'x1=x2={x1:.5f}')
    elif t>0:
        x1=(-b+math.sqrt(t))/(2*a)
        x2=(-b-math.sqrt(t))/(2*a)
        print(f'x1={x1:.5f};x2={x2:.5f}')
    else:
        re=(-b)/(2*a)
        if re==-0:
            re=0
        im=math.sqrt(-t)/(2*a)
        print(f'x1={re:.5f}+{im:.5f}i;x2={re:.5f}-{im:.5f}i')
