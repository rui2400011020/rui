string=input().lower()
n=len(string)
string=string+'0'
k=1
answer=[]
for i in range(n):
    if string[i]==string[i+1]:
        k=k+1
    else:
        answer.append(f'({string[i]},{str(k)})')
        k=1
for t in answer:
    print(t,end='')
