N=[]
P=[]
M=[]
while True:
    n,p,m=map(int,input().split())
    if n==p==m==0:
        break
    else:
        N.append(n)
        P.append(p)
        M.append(m)
for i in range(len(N)):
    n=N[i]
    p=P[i]
    m=M[i]
    answer=[]
    flag=[i for i in range(1,n+1)]
    flag=flag[p-1:]+flag[:p-1]
    for _ in range(n):
        x=m%len(flag)
        if x==0:
            x=len(flag)
        answer.append(flag[x-1])
        flag=flag[x:]+flag[:x-1]
    answer=map(str,answer)
    print(','.join(answer))
