N=[]
X=[]
while True:
    n=int(input())
    if n==0:
        break
    else:
        N.append(n)
        x=[]
        for _ in range(n):
            a,b=map(int,input().split())
            x.append([a,b])
        X.append(x)
for i in range(len(N)):
    n=N[i]
    x=X[i]
    x.sort()
    t=0
    xiao=1e5
    for i in range(n):
        if x[i][1]<xiao:
            t+=1
        xiao=min(xiao,x[i][1])
    print(t)
