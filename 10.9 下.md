P=[]
E=[]
I=[]
D=[]
while True:
    p,e,i,d=map(int,input().split())
    if p==e==i==d==-1:
        break
    else:
        P.append(p)
        E.append(e)
        I.append(i)
        D.append(d)
t=0
for j in range(len(P)):
    t+=1
    p=P[j]
    e=E[j]
    i=I[j]
    d=D[j]
    answer=(p*28*33*6+e*23*33*19+i*23*28*2-d)%21252
    if answer==0:
        answer=21252
    print(f'Case {t}: the next triple peak occurs in {answer} days.')
