s=input()
n=len(s)
m=int(input())
L=[]
R=[]
for _ in range(m):
    l,r=map(int,input().split())
    L.append(l)
    R.append(r)
flag=[0]
num=0
for i in range(n-1):
    if s[i]==s[i+1]:
        num+=1
    flag.append(num)
for i in range(m):
    l=L[i]
    r=R[i]
    answer=flag[r-1]-flag[l-1]
    print(answer)
