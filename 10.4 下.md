```
import math
N=int(input())
Answer=[]
for b in range(2,N):
    for c in range(b,N):
        for d in range(c,N):
            a=math.pow(b**3+c**3+d**3,1/3)
            t=round(a)
            if t<=N and abs(t-a)<=0.0000001:
                Answer.append([t,b,c,d])
Answer.sort()
for j in Answer:
    a=j[0]
    b=j[1]
    c=j[2]
    d=j[3]
    print(f'Cube = {a}, Triple = ({b},{c},{d})')
```
