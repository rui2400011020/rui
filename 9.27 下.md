```
n,W=map(int,input().split())
value=[]
weight=[]
price=[]
answer=0
for i in range(n):
    v,w=map(int,input().split())
    p=v/w
    value.append(v)
    weight.append(w)
    price.append([p,i])
if W>=sum(weight):
    answer=sum(value)
else:
    price.sort(reverse=True)
    t=0
    while W>=weight[price[t][1]]:
        W=W-weight[price[t][1]]
        answer+=value[price[t][1]]
        t+=1
    answer+=W*price[t][0]
print(f'{answer:.1f}')
```
