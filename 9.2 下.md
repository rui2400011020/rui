```
n=int(input())
a=list(map(int,input().split()))
b=[]
for i in a:
    i=i%2
    b.append(i)
if sum(b)==1:
    for i in range(n):
        if b[i]==1:
            c=i+1
else:
    for i in range(n):
        if b[i]==0:
            c=i+1
print(c)
```
