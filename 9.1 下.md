n=int(input())
A=[]
B=[]
C=[]
for i in range(n):
    a,b,c=map(int,input().split())
    A.append(a)
    B.append(b)
    C.append(c)
for a,b,c in zip(A,B,C):
    if a+b==c or a+c==b or b+c==a:
        print('YES')
    else:
        print('NO')
