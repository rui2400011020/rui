a=input().lower()
b=input().lower()
t=0
if a==b:
    print(0)
else:
    while True:
        c=ord(a[t])
        d=ord(b[t])
        if c==d:
            t+=1
        elif c>d:
            print(1)
            break
        elif c<d:
            print(-1)
            break
