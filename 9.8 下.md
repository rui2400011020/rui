m=int(input())
A=[]
B=[]
C=[]
D=[]
for i in range(m):
    a,b,c,d=map(int,input().split())
    A.append(a)
    B.append(b)
    C.append(c)
    D.append(d)
for i in range(m):
    a=A[i]
    b=B[i]
    c=C[i]
    d=D[i]
    t=a+b+c+d
    g=[t,t-2*a,t-2*b,t-2*c,t-2*d,t-2*a-2*b,t-2*a-2*c,t-2*a-2*d]
    if 24 in g or -24 in g:
        print('YES')
    else:
        print('NO')
