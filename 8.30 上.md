n=int(input())
a=[]
def task(t):
    s=[]
    x=[]
    a=t-t%10000
    t=t-a
    b=t-t%1000
    t=t-b
    c=t-t%100
    t=t-c
    d=t-t%10
    e=t-d
    s.append(a)
    s.append(b)
    s.append(c)
    s.append(d)
    s.append(e)
    for i in s:
        if i!=0:
            x.append(i)
    print(f'\n{len(x)}')
    for i in x:
        print(i,end=' ')
for i in range(n):
    a.append(int(input()))
for i in range(n):
    t=a[i]
    task(t)
