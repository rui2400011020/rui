import math
t=int(input())
N=list(map(int,input().split()))
nn=max(N)
n=math.ceil(math.sqrt(nn))
flag=[True]*(n+1)
prime=[]
for i in range(2,n+1):
    if flag[i]:
        prime.append(i)
    for p in prime:
        if p*i>n:
            break
        flag[p*i]=False
        if i%p==0:
            break
flag[0]=False
flag[1]=False
for x in N:
    if math.sqrt(x)%1!=0:
        print('NO')
    else:
        if flag[int(math.sqrt(x))]:
            print('YES')
        else:
            print('NO')
