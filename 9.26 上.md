m30=[4,6,9,11]
def task(year,month,day):
    day+=1
    if month in m31 and day==32:
        day=1
        month+=1
    if month in m30 and day==31:
        day=1
        month+=1
    if month==2:
        if ((year%100!=0 and year%4==0) or (year%400==0)):
            if day==30:
                day=1
                month+=1
        else:
            if day==29:
                day=1
                month+=1
    if month==13:
        month=1
        year+=1
    return year,month,day
for i in range(n):
    year,month,day=task(year,month,day)
if month<=9:
    month='0'+str(month)
if day<=9:
    day='0'+str(day)
print(f'{year}-{month}-{day}')
