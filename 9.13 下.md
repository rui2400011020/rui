a=[]
for i in range(1,100):
    if i%7!=0 and i%10!=7 and (i-i%10)!=70:
        a.append(i)
n=int(input())
b=[]
c=0
for i in a:
    if i<=n:
        b.append(i)
for i in b:
    c+=i*i
print(c)
