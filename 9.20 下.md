S=[]
while True:
    try:
        s=input()
        S.append(s)
    except EOFError:
        break
for s in S:
    n=s.count('@')
    b=s.find('@')
    c=s.count('@.')+s.count('.@')
    ss=s[b+1:]
    t=ss.count('.')
    if n==1 and s[0]!='@' and s[0]!='.' and s[-1]!='@' and s[-1]!='.' and c==0 and t>0:
        print('YES')
    else:
        print('NO')
