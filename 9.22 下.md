import math
n=int(input())
if n==0:
    print('Yes')
else:
    x=0
    t=int(math.log(n,2)//1)
    a=[0]*(t+1)
    while n>0:
        s=int(math.log(n,2)//1)
        n=n-2**s
        a[s]=1
    for i in range(t+1):
        if a[i]==a[t-i]:
            x+=1
        else:
            print('No')
            break
    if x==t+1:
        print('Yes')
