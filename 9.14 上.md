n=int(input())
A=[]
B=[]
C=[]
for i in range(n):
    a,b,c=map(int,input().split())
    A.append(a)
    B.append(b)
    C.append(c)
if sum(A)==sum(B)==sum(C)==0:
    print("YES")
else:
    print("NO")
